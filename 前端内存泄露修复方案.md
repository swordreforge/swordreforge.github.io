# 前端内存泄露修复方案(已解决)

## 📋 问题概述

本文档记录了 `index.html` 中发现的内存泄露问题及其修复方案。

---

## 🔴 高优先级问题

### 1. 事件监听器重复添加

**问题描述：**
在 `initGlobalEvents()` 函数中，使用 `removeEventListener` 后立即 `addEventListener`，但由于使用的是函数引用而非原始函数，导致无法真正移除监听器，造成监听器堆积。

**位置：**
- `initGlobalEvents()` 函数（约第 2340 行）

**当前代码：**
```javascript
function initGlobalEvents() {
    const canvasContainer = document.getElementById('canvasContainer');
    if (!canvasContainer) return;
    
    // 移除可能存在的旧监听（防止重复）
    canvasContainer.removeEventListener('mousemove', handleCanvasMouseMove);
    canvasContainer.removeEventListener('mouseup', handleCanvasMouseUp);
    canvasContainer.removeEventListener('mouseleave', handleCanvasMouseLeave);
    canvasContainer.removeEventListener('click', handleCanvasClick);
    canvasContainer.removeEventListener('mousedown', handleCanvasMouseDown);

    // 添加新监听
    canvasContainer.addEventListener('mousemove', handleCanvasMouseMove);
    canvasContainer.addEventListener('mouseup', handleCanvasMouseUp);
    canvasContainer.addEventListener('mouseleave', handleCanvasMouseLeave);
    canvasContainer.addEventListener('click', handleCanvasClick);
    canvasContainer.addEventListener('mousedown', handleCanvasMouseDown);
}
```

**问题分析：**
- `removeEventListener` 需要传入与 `addEventListener` 完全相同的函数引用
- 如果函数在不同作用域定义，即使名称相同也不是同一个引用
- 导致旧监听器无法被移除，新监听器不断添加

**修复方案：**
创建统一的事件管理器类，自动管理监听器的添加和移除。

```javascript
// ==================== 事件管理器 ====================
class EventManager {
    constructor() {
        this.listeners = new WeakMap(); // 使用 WeakMap 避免内存泄露
    }
    
    /**
     * 添加事件监听器
     * @param {Element} element - DOM 元素
     * @param {string} event - 事件名称
     * @param {Function} handler - 事件处理函数
     * @param {Object} options - 事件选项
     */
    on(element, event, handler, options) {
        // 移除旧的监听器（如果存在）
        this.off(element, event, handler);
        
        // 保存监听器引用
        if (!this.listeners.has(element)) {
            this.listeners.set(element, new Map());
        }
        const elementListeners = this.listeners.get(element);
        if (!elementListeners.has(event)) {
            elementListeners.set(event, []);
        }
        elementListeners.get(event).push(handler);
        
        // 添加监听器
        element.addEventListener(event, handler, options);
    }
    
    /**
     * 移除事件监听器
     * @param {Element} element - DOM 元素
     * @param {string} event - 事件名称
     * @param {Function} handler - 事件处理函数
     */
    off(element, event, handler) {
        if (!this.listeners.has(element)) return;
        
        const elementListeners = this.listeners.get(element);
        if (!elementListeners.has(event)) return;
        
        const handlers = elementListeners.get(event);
        const index = handlers.indexOf(handler);
        
        if (index !== -1) {
            handlers.splice(index, 1);
            element.removeEventListener(event, handler);
            
            // 如果该事件没有监听器了，清理 Map
            if (handlers.length === 0) {
                elementListeners.delete(event);
            }
        }
    }
    
    /**
     * 移除元素上的所有事件监听器
     * @param {Element} element - DOM 元素
     */
    removeAll(element) {
        if (!this.listeners.has(element)) return;
        
        const elementListeners = this.listeners.get(element);
        
        // 移除所有事件监听器
        for (const [event, handlers] of elementListeners.entries()) {
            for (const handler of handlers) {
                element.removeEventListener(event, handler);
            }
        }
        
        // 清理 Map
        this.listeners.delete(element);
    }
    
    /**
     * 移除指定事件的所有监听器
     * @param {Element} element - DOM 元素
     * @param {string} event - 事件名称
     */
    removeAllForEvent(element, event) {
        if (!this.listeners.has(element)) return;
        
        const elementListeners = this.listeners.get(element);
        if (!elementListeners.has(event)) return;
        
        const handlers = elementListeners.get(event);
        
        // 移除所有监听器
        for (const handler of handlers) {
            element.removeEventListener(event, handler);
        }
        
        // 清理 Map
        elementListeners.delete(event);
    }
}

// 创建全局事件管理器实例
const eventManager = new EventManager();

// 修改 initGlobalEvents 函数
function initGlobalEvents() {
    const canvasContainer = document.getElementById('canvasContainer');
    if (!canvasContainer) return;
    
    // 使用事件管理器添加监听器
    eventManager.on(canvasContainer, 'mousemove', handleCanvasMouseMove);
    eventManager.on(canvasContainer, 'mouseup', handleCanvasMouseUp);
    eventManager.on(canvasContainer, 'mouseleave', handleCanvasMouseLeave);
    eventManager.on(canvasContainer, 'click', handleCanvasClick);
    eventManager.on(canvasContainer, 'mousedown', handleCanvasMouseDown);
}
```

**修复后的好处：**
- 自动管理监听器，避免重复添加
- 提供统一的移除接口
- 使用 WeakMap 自动清理被销毁元素的监听器

### 2. 克隆节点导致事件监听器堆积

**问题描述：**
在 `setupWatermarkPreviewEvents()`、`setupCropEvents()`、`setupRectSelectionEvents()` 等函数中，使用 `cloneNode()` 来移除事件监听器，但原节点的事件监听器可能未被正确清理。

**位置：**
- `setupWatermarkPreviewEvents()` 函数（约第 2950 行）
- `setupCropEvents()` 函数（约第 3000 行）
- `setupRectSelectionEvents()` 函数（约第 3050 行）

**当前代码：**
```javascript
function setupWatermarkPreviewEvents() {
    const watermarkPreviewEl = document.getElementById('watermarkPreview');
    if (!watermarkPreviewEl) return;
    
    // 克隆节点以移除旧的事件监听器
    const newPreviewEl = watermarkPreviewEl.cloneNode(true);
    watermarkPreviewEl.parentNode.replaceChild(newPreviewEl, watermarkPreviewEl);
    
    // 添加新的事件监听器
    newPreviewEl.addEventListener('mousedown', handleWatermarkMouseDown);
    // ...
}
```

**问题分析：**
- `cloneNode(true)` 会复制所有子节点和事件监听器
- 但原节点的事件监听器并未被正确移除
- 可能导致事件监听器堆积和内存泄露

**修复方案：**
使用事件管理器先移除所有监听器，再添加新的。

```javascript
function setupWatermarkPreviewEvents() {

    eventManager.on(watermarkPreviewEl, 'mousedown', handleWatermarkMouseDown);
    eventManager.on(watermarkPreviewEl, 'mousemove', handleWatermarkMouseMove);
    eventManager.on(watermarkPreviewEl, 'mouseup', handleWatermarkMouseUp);
    eventManager.on(watermarkPreviewEl, 'mouseleave', handleWatermarkMouseLeave);
}

// 同样的修复应用于其他函数
```

### 3. Blob URL 未释放

**问题描述：**
在 `loadWatermark()`、`loadImage()` 等函数中，使用 `URL.createObjectURL()` 创建 Blob URL，但在不需要时未调用 `URL.revokeObjectURL()` 释放，导致内存泄露。

**位置：**
- `loadWatermark()` 函数（约第 2920 行）
- `loadImage()` 函数（约第 2530 行）
- 水印缩放滑块事件（约第 2950 行）

**当前代码：**
```javascript
async function loadWatermark(file) {
    showLoading(true);
    try {
        const arrayBuffer = await file.arrayBuffer();
        watermarkBytes = new Uint8Array(arrayBuffer);
        
        watermarkFileName.textContent = file.name;
        watermarkFileName.style.color = '#4aaa6b';
        
        document.getElementById('addWatermarkBtn').disabled = false;
        
        console.log('水印图片已加载:', file.name);
    } catch (error) {
        console.error('水印图片加载失败:', error);
        alert('水印图片加载失败');
    }
    showLoading(false);
}

// 水印缩放预览
watermarkScaleSlider.addEventListener('input', function() {
    if (!watermarkBytes) return;
    
    const tempImg = new Image();
    tempImg.onload = function() {
        const url = URL.createObjectURL(new Blob([watermarkBytes]));
        tempImg.src = url; // ❌ Blob URL 未释放
        updateWatermarkPreview();
    };
});
```

**问题分析：**
- 每次调用 `URL.createObjectURL()` 都会创建一个新的 URL
- 如果不调用 `URL.revokeObjectURL()`，这些 URL 会一直占用内存
- 频繁创建不释放会导致严重的内存泄露

**修复方案：**
创建 Blob URL 管理器，自动跟踪和释放 Blob URL。

```javascript
// ==================== Blob URL 管理器 ====================
class BlobURLManager {
    constructor() {
        this.urls = new Set(); // 存储所有创建的 Blob URL
    }
    
    /**
     * 创建 Blob URL 并跟踪
     * @param {Blob} blob - Blob 对象
     * @returns {string} Blob URL
     */
    create(blob) {
        const url = URL.createObjectURL(blob);
        this.urls.add(url);
        return url;
    }
    
    /**
     * 释放 Blob URL
     * @param {string} url - Blob URL
     */
    revoke(url) {
        if (this.urls.has(url)) {
            URL.revokeObjectURL(url);
            this.urls.delete(url);
        }
    }
    
    /**
     * 释放所有 Blob URL
     */
    revokeAll() {
        this.urls.forEach(url => {
            URL.revokeObjectURL(url);
        });
        this.urls.clear();
    }
    
    /**
     * 获取当前管理的 Blob URL 数量
     * @returns {number}
     */
    getCount() {
        return this.urls.size;
    }
}

// 创建全局 Blob URL 管理器实例
const blobURLManager = new BlobURLManager();

// 修改 loadWatermark 函数
async function loadWatermark(file) {
    showLoading(true);
    try {
        const arrayBuffer = await file.arrayBuffer();
        watermarkBytes = new Uint8Array(arrayBuffer);
        
        watermarkFileName.textContent = file.name;
        watermarkFileName.style.color = '#4aaa6b';
        
        document.getElementById('addWatermarkBtn').disabled = false;
        
        console.log('水印图片已加载:', file.name);
    } catch (error) {
        console.error('水印图片加载失败:', error);
        alert('水印图片加载失败');
    }
    showLoading(false);
}

// 修改水印缩放预览
watermarkScaleSlider.addEventListener('input', function() {
    if (!watermarkBytes) return;
    
    const tempImg = new Image();
    const blob = new Blob([watermarkBytes]);
    const url = blobURLManager.create(blob); // 使用管理器创建 URL
    
    tempImg.onload = function() {
        tempImg.src = url;
        updateWatermarkPreview();
        blobURLManager.revoke(url); // 使用完成后释放
    };
    
    tempImg.onerror = function() {
        blobURLManager.revoke(url); // 错误时也要释放
    };
});

// 同样的修复应用于 loadImage 函数
function loadImage(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = function(e) {
            const img = new Image();
            img.onload = function() {
                const blob = new Blob([e.target.result]);
                const url = blobURLManager.create(blob);
                img.src = url;
                resolve(img);
                blobURLManager.revoke(url);
            };
            img.onerror = function() {
                reject(new Error('Failed to load image'));
            };
            img.src = blobURLManager.create(new Blob([e.target.result]));
        };
        reader.onerror = function() {
            reject(new Error('Failed to read file'));
        };
        reader.readAsArrayBuffer(file);
    });
}

// 在页面卸载时释放所有 Blob URL
window.addEventListener('beforeunload', () => {
    blobURLManager.revokeAll();
});
```

### 4. IndexedDB 连接未复用

**问题描述：**
在 `initFontDB()`、`saveFontToDB()`、`restoreFontsFromDB()` 等函数中，每次操作都打开新的 IndexedDB 连接，但没有复用或关闭连接，导致连接堆积。

**位置：**
- `initFontDB()` 函数（约第 2300 行）
- `saveFontToDB()` 函数（约第 2350 行）
- `restoreFontsFromDB()` 函数（约第 2400 行）

**当前代码：**
```javascript
// 初始化 IndexedDB
async function initFontDB() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, DB_VERSION);
        
        request.onerror = () => reject(request.error);
        request.onsuccess = () => resolve(request.result);
        
        request.onupgradeneeded = (event) => {
            const db = event.target.result;
            if (!db.objectStoreNames.contains(STORE_NAME)) {
                db.createObjectStore(STORE_NAME, { keyPath: 'name' });
            }
        };
    });
}

// 保存字体到数据库
async function saveFontToDB(name, data) {
    try {
        const db = await initFontDB(); // ❌ 每次都打开新连接
        const transaction = db.transaction(STORE_NAME, 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        await new Promise((resolve, reject) => {
            const request = store.put({ name, data });
            request.onsuccess = resolve;
            request.onerror = () => reject(request.error);
        });
    } catch (error) {
        console.error('保存字体失败:', error);
    }
}
```

**问题分析：**
- 每次调用 `initFontDB()` 都会打开新的数据库连接
- 未复用连接，也未在不需要时关闭连接
- 可能导致连接数过多，影响性能

**修复方案：**
使用单例模式管理 IndexedDB 连接。

```javascript
// ==================== IndexedDB 连接管理 ====================
class FontDBManager {
    constructor() {
        this.db = null;
        this.initPromise = null;
    }
    
    /**
     * 获取数据库连接（单例）
     * @returns {Promise<IDBDatabase>}
     */
    async getConnection() {
        if (this.db && this.db.readyState !== 'closed') {
            return this.db;
        }
        
        if (!this.initPromise) {
            this.initPromise = this._initDB();
        }
        
        return this.initPromise;
    }
    
    /**
     * 初始化数据库
     * @returns {Promise<IDBDatabase>}
     */
    async _initDB() {
        return new Promise((resolve, reject) => {
            const request = indexedDB.open(DB_NAME, DB_VERSION);
            
            request.onerror = () => {
                this.initPromise = null;
                reject(request.error);
            };
            
            request.onsuccess = () => {
                this.db = request.result;
                // 监听数据库关闭事件
                this.db.onclose = () => {
                    this.db = null;
                    this.initPromise = null;
                };
                resolve(this.db);
            };
            
            request.onupgradeneeded = (event) => {
                const db = event.target.result;
                if (!db.objectStoreNames.contains(STORE_NAME)) {
                    db.createObjectStore(STORE_NAME, { keyPath: 'name' });
                }
            };
        });
    }
    
    /**
     * 保存字体到数据库
     * @param {string} name - 字体名称
     * @param {Uint8Array} data - 字体数据
     * @returns {Promise<void>}
     */
    async saveFont(name, data) {
        try {
            const db = await this.getConnection();
            const transaction = db.transaction(STORE_NAME, 'readwrite');
            const store = transaction.objectStore(STORE_NAME);
            
            await new Promise((resolve, reject) => {
                const request = store.put({ name, data });
                request.onsuccess = resolve;
                request.onerror = () => reject(request.error);
            });
        } catch (error) {
            console.error('保存字体失败:', error);
            throw error;
        }
    }
    
    /**
     * 从数据库删除字体
     * @param {string} name - 字体名称
     * @returns {Promise<void>}
     */
    async deleteFont(name) {
        try {
            const db = await this.getConnection();
            const transaction = db.transaction(STORE_NAME, 'readwrite');
            const store = transaction.objectStore(STORE_NAME);
            
            await new Promise((resolve, reject) => {
                const request = store.delete(name);
                request.onsuccess = resolve;
                request.onerror = () => reject(request.error);
            });
        } catch (error) {
            console.error('删除字体失败:', error);
            throw error;
        }
    }
    
    /**
     * 从数据库恢复所有字体
     * @returns {Promise<Array<{name: string, data: Uint8Array}>>}
     */
    async getAllFonts() {
        try {
            const db = await this.getConnection();
            const transaction = db.transaction(STORE_NAME, 'readonly');
            const store = transaction.objectStore(STORE_NAME);
            
            return new Promise((resolve, reject) => {
                const request = store.getAll();
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        } catch (error) {
            console.error('获取字体失败:', error);
            throw error;
        }
    }
    
    /**
     * 清空所有字体
     * @returns {Promise<void>}
     */
    async clearAllFonts() {
        try {
            const db = await this.getConnection();
            const transaction = db.transaction(STORE_NAME, 'readwrite');
            const store = transaction.objectStore(STORE_NAME);
            
            await new Promise((resolve, reject) => {
                const request = store.clear();
                request.onsuccess = resolve;
                request.onerror = () => reject(request.error);
            });
        } catch (error) {
            console.error('清空字体失败:', error);
            throw error;
        }
    }
    
    /**
     * 关闭数据库连接
     */
    close() {
        if (this.db && this.db.readyState !== 'closed') {
            this.db.close();
            this.db = null;
            this.initPromise = null;
        }
    }
}

// 创建全局字体数据库管理器实例
const fontDBManager = new FontDBManager();

// 修改相关函数以使用管理器
async function saveFontToDB(name, data) {
    await fontDBManager.saveFont(name, data);
}

async function restoreFontsFromDB() {
    try {
        const fonts = await fontDBManager.getAllFonts();
        const registeredFontsList = document.getElementById('registeredFontsList');
        
        if (fonts.length === 0) {
            registeredFontsList.innerHTML = '<div style="color: #666; font-size: 0.85em;">暂无自定义字体</div>';
            document.getElementById('clearFontsBtn').disabled = true;
            return;
        }
        
        registeredFontsList.innerHTML = '';
        for (const font of fonts) {
            try {
                await wasm_register_font(font.name, font.data);
                addFontToUI(font.name);
            } catch (error) {
                console.error(`注册字体失败: ${font.name}`, error);
            }
        }
        
        document.getElementById('clearFontsBtn').disabled = false;
    } catch (error) {
        console.error('恢复字体失败:', error);
    }
}

async function clearCustomFonts() {
    if (confirm('确定要清空所有自定义字体吗？')) {
        try {
            await fontDBManager.clearAllFonts();
            const registeredFontsList = document.getElementById('registeredFontsList');
            registeredFontsList.innerHTML = '<div style="color: #666; font-size: 0.85em;">暂无自定义字体</div>';
            document.getElementById('clearFontsBtn').disabled = true;
            alert('自定义字体已清空');
        } catch (error) {
            console.error('清空字体失败:', error);
            alert('清空字体失败');
        }
    }
}

// 在页面卸载时关闭数据库连接
window.addEventListener('beforeunload', () => {
    fontDBManager.close();
});
```

### 5. 全局变量堆积

**问题描述：**
代码中大量使用全局变量（`isBrushMode`、`isWatermarkMode`、`isTextMode`、`isCropMode` 等），在模块关闭时不会清理，导致内存占用持续增长。

**位置：**
- 整个 `<script type="module">` 部分（约第 2100 行开始）

**当前代码：**
```javascript
// 全局变量声明
let wasmModule = null;
let processor = null;
let originalImageData = null;
let currentImage = null;

let offscreenCanvas = null;
let offscreenCtx = null;

let historyStack = [];
let redoStack = [];

let isBrushMode = false;
let isWatermarkMode = false;
let isTextMode = false;
let isCropMode = false;
let isPolygonCropMode = false;
let isRectSelectionMode = false;

let brushSettings = {
    color: '#000000',
    width: 5,
    type: 'basic'
};

let polygonPoints = [];
let cropBox = { x: 0, y: 0, width: 0, height: 0 };
let watermarkBytes = null;
let blendBytes = null;

// ... 更多全局变量
```

**问题分析：**
- 大量全局变量在应用生命周期中一直占用内存
- 没有清理机制，即使不再使用也不会释放
- 长时间使用会导致内存占用持续增长

**修复方案：**
创建统一的状态管理器和清理函数。

```javascript
// ==================== 应用状态管理 ====================
class AppState {
    constructor() {
        // 核心对象
        this.wasmModule = null;
        this.processor = null;
        this.originalImageData = null;
        this.currentImage = null;
```

### 6. Image 对象未释放

**问题描述：**
在 `loadImage()`、`loadWatermark()` 等函数中创建的 `Image` 对象在 onload 后没有置 null，导致对象无法被垃圾回收。

**位置：**
- `loadImage()` 函数（约第 2530 行）
- `loadWatermark()` 函数（约第 2920 行）

**修复方案：**

```javascript
// 创建 Image 对象管理器
class ImageManager {
    constructor() {
        this.images = new Map(); // 存储 Image 对象
    }
    
    /**
     * 创建 Image 对象并跟踪
     * @param {string} src - 图片源
     * @returns {HTMLImageElement}
     */
    create(src) {
        const img = new Image();
        img.onload = () => {
            // 图片加载完成后，从管理器中移除
            this.images.delete(img);
        };
        img.onerror = () => {
            // 加载失败时，也从管理器中移除
            this.images.delete(img);
        };
        img.src = src;
        this.images.set(img, true);
        return img;
    }
    
    /**
     * 销毁 Image 对象
     * @param {HTMLImageElement} img - Image 对象
     */
    destroy(img) {
        if (this.images.has(img)) {
            img.onload = null;
            img.onerror = null;
            img.src = '';
            this.images.delete(img);
        }
    }
    
    /**
     * 销毁所有 Image 对象
     */
    destroyAll() {
        this.images.forEach((_, img) => {
            img.onload = null;
            img.onerror = null;
            img.src = '';
        });
        this.images.clear();
    }
}

// 创建全局 Image 管理器实例
const imageManager = new ImageManager();

// 修改 loadImage 函数
function loadImage(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = function(e) {
            const blob = new Blob([e.target.result]);
            const url = blobURLManager.create(blob);
            const img = imageManager.create(url);
            
            img.onload = function() {
                resolve(img);
                blobURLManager.revoke(url);
            };
            
            img.onerror = function() {
                blobURLManager.revoke(url);
                reject(new Error('Failed to load image'));
            };
        };
        reader.onerror = function() {
            reject(new Error('Failed to read file'));
        };
        reader.readAsArrayBuffer(file);
    });
}
```

```javascript
// 创建定时器管理器
class TimerManager {
    constructor() {
        this.timers = new Set(); // 存储所有定时器 ID
    }
    
    /**
     * 设置定时器并跟踪
     * @param {Function} callback - 回调函数
     * @param {number} delay - 延迟时间
     * @returns {number} 定时器 ID
     */
    setTimeout(callback, delay) {
        const id = window.setTimeout(() => {
            this.timers.delete(id);
            callback();
        }, delay);
        this.timers.add(id);
        return id;
    }
    
    /**
     * 清除定时器
     * @param {number} id - 定时器 ID
     */
    clearTimeout(id) {
        window.clearTimeout(id);
        this.timers.delete(id);
    }
    
    /**
     * 清除所有定时器
     */
    clearTimeoutAll() {
        this.timers.forEach(id => {
            window.clearTimeout(id);
        });
        this.timers.clear();
    }
}
```

```javascript
// 使用定时器管理器中的 Set 来管理 resize 监听器
const resizeHandlers = new Map(); // Map<element, Set<Function>>

function addResizeHandler(element, handler) {
    if (!resizeHandlers.has(element)) {
        resizeHandlers.set(element, new Set());
    }
    resizeHandlers.get(element).add(handler);
    window.addEventListener('resize', handler);
}

function removeResizeHandler(element, handler) {
    if (resizeHandlers.has(element)) {
        const handlers = resizeHandlers.get(element);
        if (handlers.has(handler)) {
            window.removeEventListener('resize', handler);
            handlers.delete(handler);
        }
    }
}

function removeAllResizeHandlers() {
    resizeHandlers.forEach((handlers, element) => {
        handlers.forEach(handler => {
            window.removeEventListener('resize', handler);
        });
    });
    resizeHandlers.clear();
}

// 修改 cropImage 函数
window.cropImage = function() {
    // ...
    const resizeHandler = () => {
        if (isCropMode) {
            // ...
        }
    };
    addResizeHandler(canvas, resizeHandler);
    window._cropResizeHandler = resizeHandler;
};

// 修改 cancelCrop 函数
window.cancelCrop = function() {
    // ...
    if (window._cropResizeHandler) {
        removeResizeHandler(canvas, window._cropResizeHandler);
        delete window._cropResizeHandler;
    }
};

// 在页面卸载时清理所有 resize 监听器
```

```javascript
let pendingDOMUpdates = {};
let domUpdateScheduled = false;

function scheduleDOMUpdate(id, value) {
    pendingDOMUpdates[id] = value;
    
    if (!domUpdateScheduled) {
        domUpdateScheduled = true;
```
