  字体持久化方案

  方案概述
  使用 IndexedDB 存储上传的字体数据，实现字体持久化。IndexedDB
  是浏览器内置的数据库，可以存储大量二进制数据，并且容量比 localStorage
  大得多。

  实现步骤

  1. 创建 IndexedDB 管理模块
      // IndexedDB 配置
      const DB_NAME = 'PhotonFontDB';
      const DB_VERSION = 1;
      const STORE_NAME = 'fonts';

   // 初始化 IndexedDB
   async function initFontDB() {
       return new Promise((resolve, reject) => {
           const request = indexedDB.open(DB_NAME, DB_VERSION);

           request.onerror = () => reject(request.error);
           request.onsuccess = () => resolve(request.result);
    
           request.onupgradeneeded = (event) => {
               const db = event.target.result;
               if (!db.objectStoreNames.contains(STORE_NAME)) {
                   db.createObjectStore(STORE_NAME, { keyPath:
   'name' });
               }
           };
       });
   }

   // 保存字体到 IndexedDB
   async function saveFontToDB(name, data) {
       const db = await initFontDB();
       return new Promise((resolve, reject) => {
           const transaction = db.transaction([STORE_NAME],
   'readwrite');
           const store = transaction.objectStore(STORE_NAME);
           const request = store.put({ name, data, timestamp: Date.
   now() });

           request.onsuccess = () => resolve();
           request.onerror = () => reject(request.error);
       });
   }

   // 从 IndexedDB 加载所有字体
   async function loadFontsFromDB() {
       const db = await initFontDB();
       return new Promise((resolve, reject) => {
           const transaction = db.transaction([STORE_NAME],
   'readonly');
           const store = transaction.objectStore(STORE_NAME);
           const request = store.getAll();

           request.onsuccess = () => resolve(request.result);
           request.onerror = () => reject(request.error);
       });
   }

   // 从 IndexedDB 删除字体
   async function deleteFontFromDB(name) {
       const db = await initFontDB();
       return new Promise((resolve, reject) => {
           const transaction = db.transaction([STORE_NAME],
   'readwrite');
           const store = transaction.objectStore(STORE_NAME);
           const request = store.delete(name);

           request.onsuccess = () => resolve();
           request.onerror = () => reject(request.error);
       });
   }

   // 清空 IndexedDB 中的所有字体
   async function clearFontsFromDB() {
       const db = await initFontDB();
       return new Promise((resolve, reject) => {
           const transaction = db.transaction([STORE_NAME],
   'readwrite');
           const store = transaction.objectStore(STORE_NAME);
           const request = store.clear();

           request.onsuccess = () => resolve();
           request.onerror = () => reject(request.error);
       });
   }

  2. 在页面加载时恢复字体
    在 WASM 模块初始化完成后，从 IndexedDB 加载字体并注册到 WASM：

   // 在 loadWasmModule() 函数返回后调用
   async function restoreFontsFromDB() {
       try {
           const fonts = await loadFontsFromDB();

           for (const font of fonts) {
               try {
                   // 注册字体到 WASM
                   wasm_register_font(font.name, font.data);
    
                   // 添加到字体选择下拉框
                   addCustomFontToSelect(font.name);
    
                   console.log(`字体 "${font.name}" 已恢复`);
               } catch (error) {
                   console.error(`恢复字体 "${font.name}" 失败:`,
   error);
               }
           }

           // 更新字体列表显示
           updateRegisteredFontsList();
    
           if (fonts.length > 0) {
               console.log(`已恢复 ${fonts.length} 个字体`);
           }
       } catch (error) {
           console.error('从数据库恢复字体失败:', error);
       }
   }

  3. 在上传字体时保存到 IndexedDB
    修改 applyUploadedFont 函数：

   window.applyUploadedFont = async function() {
       if (!uploadedFontData || !uploadedFontName) {
           alert('请先上传字体文件');
           return;
       }

       const fontNameToRegister = uploadedFontName;
    
       try {

           if (wasm_is_font_registered(uploadedFontName)) {

                   return;
               }
           }

           // 注册字体到 WASM
           wasm_register_font(uploadedFontName, uploadedFontData);
    
           // 保存到 IndexedDB（持久化）
           await saveFontToDB(uploadedFontName, uploadedFontData);
    
           // 更新字体列表显示
           updateRegisteredFontsList();

           // 选中这个新字体

   );

           // 清空上传数据
           uploadedFontData = null;
           uploadedFontName = null;
           fontFileName.textContent = '点击上传自定义字体';
           fontFileName.style.color = '#888';
           document.getElementById('applyFontBtn').disabled = true;

       } catch (error) {

       }

  4. 在删除字体时从 IndexedDB 删除
    修改 removeFont 函数：

   window.removeFont = async function(fontName) {
       if (!confirm(`确定要删除字体 "${fontName}" 吗？`)) {
           return;
       }

       try {
           // 从 WASM 注销字体
           wasm_unregister_font(fontName);
    
           // 从 IndexedDB 删除
           await deleteFontFromDB(fontName);
    
           // 从下拉框中移除
           const fontSelect = document.getElementById('fontSelect'
   );
           const option = fontSelect.querySelector(
   `option[value="custom-${fontName}"]`);
           if (option) {
               option.remove();
           }

           // 更新字体列表
           updateRegisteredFontsList();
    
           alert(`字体 "${fontName}" 已删除`);
       } catch (error) {
           console.error('删除字体失败:', error);
           alert(`删除字体失败: ${error.message}`);
       }
   };

  5. 在清空字体时清空 IndexedDB
    修改 clearCustomFonts 函数：

   window.clearCustomFonts = async function() {
       if (!confirm('确定要清空所有自定义字体吗？此操作不可恢复。'
   )) {
           return;
       }

       try {
           // 清空 WASM 中的字体
           wasm_clear_fonts();
    
           // 清空 IndexedDB
           await clearFontsFromDB();
    
           // 从下拉框中移除所有自定义字体选项
           const fontSelect = document.getElementById('fontSelect'
   );
           const customOptions = fontSelect.querySelectorAll(
   'option[value^="custom-"]');
           customOptions.forEach(option => option.remove());

           // 重置为默认选项
           fontSelect.value = 'default';
    
           // 更新字体列表
           updateRegisteredFontsList();
    
           alert('所有自定义字体已清空');
       } catch (error) {
           console.error('清空字体失败:', error);
           alert(`清空字体失败: ${error.message}`);
       }
   };

  优点
   1. 持久化：字体数据存储在浏览器数据库中，刷新页面后仍然可用
   2. 容量大：IndexedDB 容量远大于 localStorage，可以存储多个字体文件
   3. 性能好：IndexedDB 是异步的，不会阻塞主线程
   4. 兼容性好：所有现代浏览器都支持 IndexedDB

  注意事项
   1. 浏览器限制：不同浏览器的 IndexedDB
      容量限制不同，但通常足够存储多个字体文件
   2. 隐私模式：在隐私/无痕模式下，IndexedDB 会在关闭浏览器后被清除
   3. 跨域：IndexedDB 数据只对同一域名可用
   4. 清理缓存：如果用户清除浏览器数据，字体也会被清除

✦ 这个方案可以满足用户的需求：只要上传一次字体，不清理浏览器缓存或在侧边
  栏删除字体，字体就会一直可用。
