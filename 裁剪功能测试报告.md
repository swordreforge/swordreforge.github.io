// 第3941-3969行
window.cropImage = function() {
    if (!processor) return;
    isCropMode = true;

    // 初始化裁剪框为图像中心的一半大小
    const canvas = document.getElementById('imageCanvas');
    const width = canvas.width;
    const height = canvas.height;
    cropBox = {
        x: width * 0.25,
        y: height * 0.25,
        width: width * 0.5,
        height: height * 0.5
    };

    // 显示裁剪覆盖层
    const cropOverlay = document.getElementById('cropOverlay');
    cropOverlay.style.display = 'block';
    cropOverlay.width = width;
    cropOverlay.height = height;

    // 显示裁剪控制按钮
    document.getElementById('cropControls').style.display = 'flex';

    // 绘制裁剪框
    drawCropOverlay();

    // 添加事件监听

};

### B. 绘制裁剪框
```javascript
// 第4116-4174行
function drawCropOverlay() {
    const cropOverlay = document.getElementById('cropOverlay');
    const ctx = cropOverlay.getContext('2d');

    // 清空覆盖层
    ctx.clearRect(0, 0, cropOverlay.width, cropOverlay.height);

    // 绘制半透明遮罩
    ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
    ctx.fillRect(0, 0, cropOverlay.width, cropOverlay.height);

    // 清除裁剪框区域的遮罩
    ctx.clearRect(cropBox.x, cropBox.y, cropBox.width, cropBox.height);

    // 绘制裁剪框边框
    ctx.strokeStyle = '#fff';
    ctx.lineWidth = 2;
    ctx.setLineDash([5, 5]);
    ctx.strokeRect(cropBox.x, cropBox.y, cropBox.width, cropBox.height);
    ctx.setLineDash([]);

    // 绘制三分线
    ctx.strokeStyle = 'rgba(255, 255, 255, 0.3)';
    ctx.lineWidth = 1;

    // 垂直线
    for (let i = 1; i < 3; i++) {
        ctx.beginPath();
        ctx.moveTo(cropBox.x + (cropBox.width / 3) * i, cropBox.y);
        ctx.lineTo(cropBox.x + (cropBox.width / 3) * i, cropBox.y + cropBox.height);
        ctx.stroke();
    }

    // 水平线
    for (let i = 1; i < 3; i++) {
        ctx.beginPath();
        ctx.moveTo(cropBox.x, cropBox.y + (cropBox.height / 3) * i);
        ctx.lineTo(cropBox.x + cropBox.width, cropBox.y + (cropBox.height / 3) * i);
        ctx.stroke();
    }

    // 绘制调整手柄
    const handleSize = 16;
    ctx.fillStyle = '#fff';
    ctx.strokeStyle = '#333';
    ctx.lineWidth = 2;

    // 四角
    ctx.fillRect(cropBox.x - handleSize/2, cropBox.y - handleSize/2, handleSize, handleSize);
    ctx.strokeRect(cropBox.x - handleSize/2, cropBox.y - handleSize/2, handleSize, handleSize);

    ctx.fillRect(cropBox.x + cropBox.width - handleSize/2, cropBox.y - handleSize/2, handleSize, handleSize);
    ctx.strokeRect(cropBox.x + cropBox.width - handleSize/2, cropBox.y - handleSize/2, handleSize, handleSize);

    ctx.fillRect(cropBox.x - handleSize/2, cropBox.y + cropBox.height - handleSize/2, handleSize, handleSize);
    ctx.strokeRect(cropBox.x - handleSize/2, cropBox.y + cropBox.height - handleSize/2, handleSize, handleSize);

    ctx.fillRect(cropBox.x + cropBox.width - handleSize/2, cropBox.y + cropBox.height - handleSize/2, handleSize, handleSize);
    ctx.strokeRect(cropBox.x + cropBox.width - handleSize/2, cropBox.y + cropBox.height - handleSize/2, handleSize, handleSize);
}
```
